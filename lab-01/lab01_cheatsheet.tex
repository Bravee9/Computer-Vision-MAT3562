\section{Đọc, Ghi và Hiển thị Ảnh (OpenCV)}
\begin{itemize}
    \item \texttt{cv2.imread(filepath, flag)}: Đọc ảnh. Flag \texttt{cv2.IMREAD\_COLOR} (mặc định BGR), \texttt{cv2.IMREAD\_GRAYSCALE} (ảnh xám).
    \item \texttt{cv2.imwrite(filepath, img)}: Lưu ảnh ra file.
    \item \texttt{cv2.imshow(winname, img)}: Hiển thị ảnh (kèm \texttt{cv2.waitKey(0)} để dừng màn hình).
    \item Chuyển đổi màu: \texttt{cv2.cvtColor(img, cv2.COLOR\_BGR2RGB)} (để hiển thị đúng bằng Matplotlib).
\end{itemize}

\section{Numpy trong Xử lý ảnh}
\begin{itemize}
    \item Kích thước ảnh: \texttt{h, w, c = img.shape}.
    \item Loại dữ liệu: \texttt{img.dtype} (thường là \texttt{uint8} [0, 255] hoặc \texttt{float32} [0, 1]).
    \item Truy xuất pixel: \texttt{img[y, x]} hoặc \texttt{img[y, x, channel]}.
    \item Cắt vùng ảnh (Crop): \texttt{roi = img[y1:y2, x1:x2]}.
\end{itemize}
